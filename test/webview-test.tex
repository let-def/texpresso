  \nabla \cdot \mathbf{E} &= \frac{\rho}{\varepsilon_{0}} \\
  \nabla \cdot \mathbf{B} &= 0 \\
  \nabla \times \mathbf{E} &= -\frac{\partial \mathbf{B}}{\partial t} \\
  \nabla \times \mathbf{B} &= \mu_{0}\mathbf{J} +

    """Generate first n Fibonacci numbers."""
    result = [0, 1]
    for i in range(2, n):

    return result[:n]
